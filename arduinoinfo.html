Parts Used

  • Jumper Wires x 17
  • Arduino Nano ESP32 x 1
  • LEDs x 4 (any color)
  • small button x 1
  • 2 way switch I/O x 1 (optional)
  • ~5V battery pack (optional)
            
// declaring constants                    
const int ledpins[] = {D3,D5,D9,D6};
const int pinsSize = 4;

int i,lastState;

void setup() {
  // setting pins connected to leds to output
  for(i = 0; i < pinsSize; i++){
    pinMode(ledpins[i],OUTPUT);
  }
  // setting button and switch as inputs (using pullup resistor)
  pinMode(D7,INPUT_PULLUP);
  pinMode(D2,INPUT_PULLUP);
  lastState = digitalRead(D2);
}

// read button's current state
int currentState = digitalRead(D2);

void loop() {
  if(digitalRead(D7) == HIGH) { return; }

  currentState = digitalRead(D2);

  // sequence (extra when button is pressed)
  if(currentState != lastState){
    for(i = 0; i < pinsSize; i++){
      analogWrite(ledpins[i],150);
    }
    delay(1000);
    for(i = 0; i < pinsSize; i++){
      analogWrite(ledpins[i],0);
    }
    delay(1000);
    for(i = 0; i < pinsSize; i++){
      analogWrite(ledpins[i],150);
    }
    delay(1000);
    for(i = 0; i < pinsSize; i++){
      analogWrite(ledpins[i],0);
    }
    delay(1000);
    lastState = currentState;
    return;
  }

  // normal sequence (flash)
  for(i = 0; i < pinsSize; i++){
    analogWrite(ledpins[i],150);
    delay(100);
    analogWrite(ledpins[i],0);
  }

  for(i = 0; i < 1; i++){
    analogWrite(D3,150);
    delay(100);
    analogWrite(D3,0);
  }

  for(i = 0; i < 1; i++){
    analogWrite(D6,150);
    delay(100);
    analogWrite(D6,0);
  }

  for(i = pinsSize; i > 0; i--){
    analogWrite(ledpins[i],150);
    delay(100);
    analogWrite(ledpins[i],0);
  }
}
